class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        left, right = 0, len(nums) - 1
        lower = -1

        while left <= right:
            middle = (left + right) // 2

            if target == nums[middle]:
                lower = middle
                right = middle - 1
            elif target < nums[middle]:
                right = middle - 1
            else:
                left = middle + 1
        
        upper = -1

        left, right = 0, len(nums) - 1
        while left <= right:
            middle = (left + right) // 2

            if target == nums[middle]:
                upper = middle
                left = middle + 1
            elif target < nums[middle]:
                right = middle - 1
            else:
                left = middle + 1

        return [lower, upper]
