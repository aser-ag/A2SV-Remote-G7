class Solution:
    def simplifyPath(self, path: str) -> str:
        split = path.split("/")
        
        stack = []

        for directory in split:
            if directory == "..":
                if len(stack) > 0:
                    stack.pop()
            elif directory == "." or directory == "":
                continue
            else:
                stack.append(directory)

        res = "/" + "/".join(stack)
        return res
