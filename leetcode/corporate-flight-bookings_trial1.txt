class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        diff = [0] * (n + 1)
        # 0, 0, 0, 0, 0, 0

        for first, last, seats in bookings:
            diff[first - 1] += seats
            diff[last] -= seats
        
        res = [diff[0]] * n
        for i in range(1, n):
            res[i] = res[i - 1] + diff[i]
        
        return res
