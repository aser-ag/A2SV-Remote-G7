class Solution:
    def intervalIntersection(self, firstList: List[List[int]], secondList: List[List[int]]) -> List[List[int]]:
        i, j = 0, 0
        result = []
        
        while i < len(firstList) and j < len(secondList):
            a, b = firstList[i]
            c, d = secondList[j]
            
            # Find overlap
            start = max(a, c)
            end = min(b, d)
            
            if start <= end:
                result.append([start, end])
            
            # Move pointer
            if b < d:
                i += 1
            else:
                j += 1
        
        return result
